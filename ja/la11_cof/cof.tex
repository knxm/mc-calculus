\element{cof}{
\begin{question}{cof01}
行列
\(
A=\begin{pmatrix}
 1 & -2 & -1\\
 -2 & 4 & 1\\
 -3 & 5 & 3
\end{pmatrix}
\) 
の余因子行列\(\widetilde{A}\)を求めなさい．
% 7, 1, 2
% 3, 0, 1
% 2, 1, 0
\AMCOpen{lines=4, dots=false, framerule=0pt}{\wrongchoice[W]{誤}\scoring{0}\wrongchoice[P]{部}\scoring{1}\correctchoice[C]{正}\scoring{2}}
\end{question}
}
