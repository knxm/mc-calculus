\begin{luacode*}
function execMaxima(command)
  local maxima_command="echo 'tex1("..command..");' | maxima --very-quiet "
  local handle=io.popen(maxima_command, "r")
  local content=string.gsub(handle:read("*all"), "\n", "")
  handle:close()
  return content
end
